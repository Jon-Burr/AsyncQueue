78 std::string MessageFormatter::formatTime(

80 auto seconds = std::chrono::floor<std::chrono::seconds>(timepoint);
81 if (format.find("%+") != std::string::npos) {
82 std::size_t pos = format.find("%+");
83 auto diff = timepoint - seconds;
84 while (pos != std::string::npos) {
85 if (pos + 2 >= format.size())
86 break;
87 char code = format.at(pos + 2);
88 std::size_t count;
89 if (code == 'n')
90 count = std::chrono::duration_cast<std::chrono::nanoseconds>(diff).count();
91 else if (code == 'u')
92 count = std::chrono::duration_cast<std::chrono::microseconds>(diff).count();
93 else if (code == 'm')
94 count = std::chrono::duration_cast<std::chrono::milliseconds>(diff).count();
95 else
96 break;
97 format = format.replace(pos, 3, std::to_string(count));
98 pos = format.find("%+", pos);
99 }
100 }
101 std::ostringstream oss;
102 std::time_t tmt = Clock::to_time_t(seconds);
103 oss << std::put_time(std::localtime(&tmt), format.c_str());
104 return oss.str();
